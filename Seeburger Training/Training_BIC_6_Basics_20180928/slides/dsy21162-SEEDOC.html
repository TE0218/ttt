Illustration

  1. Changing the code for using the MapDB commands.

 
// Open DB-Connection
connectDB("DB","jdbc:seeburger:jucon://DS_MD","","",
"com.seeburger.jucon.locator.DataSourceDriver","off");

// Open MapDB Channels
openMapDB("EAN","DB","INTERNAL_ARTICLE_NUMBER_TO_EAN",
          "cField1","cField2","");
openMapDB("description","DB","INTERNAL_ARTICLE_NUMBER_TO_EAN",
          "cField1","cField3","");

local internal_article_number$;
local EAN$;
local description$;
internal_article_number$ = "123546";

// Use MapDB command
copy MapDB("EAN",internal_article_number$) to EAN$;
copy MapDB("description",internal_article_number$)  to description$;


traceln("EAN         : " & EAN$);
traceln("Description : " & description$);
